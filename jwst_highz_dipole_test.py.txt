COLD_SPOT_RA = 55.0   # Eridanus Supervoid (Minimum H0 expected)
COLD_SPOT_DEC = -25.0
H0_BARE = 67.4        # Planck Baseline
OMEGA_M = 0.315       # Background matter density

def angular_separation(ra1, dec1, ra2, dec2):
    """Haversine formula for angular distance on celestial sphere."""
    ra1, dec1 = np.radians(ra1), np.radians(dec1)
    ra2, dec2 = np.radians(ra2), np.radians(dec2)
    dlon = ra2 - ra1
    dlat = dec2 - dec1
    a = np.sin(dlat/2)**2 + np.cos(dec1) * np.cos(dec2) * np.sin(dlon/2)**2
    return np.degrees(2 * np.arcsin(np.sqrt(np.clip(a, 0, 1))))

def theoretical_distance_modulus(z, h0, omega_m):
    """Approximate distance modulus for flat LCDM background."""
    from scipy.integrate import quad
    def integrand(x):
        return 1.0 / np.sqrt(omega_m * (1+x)**3 + (1-omega_m))
    
    mu_theoretical = []
    for z_i in z:
        integral, _ = quad(integrand, 0, z_i)
        dL = (299792.458 * (1 + z_i) / h0) * integral
        mu_theoretical.append(5 * np.log10(dL) + 25)
    return np.array(mu_theoretical)

def dipole_model(theta, h_iso, a_dip):
    """Cosine dipole model for continuous fitting: H(theta) = H_iso + A_dip * cos(theta)"""
    return h_iso + a_dip * np.cos(np.radians(theta))

def generate_mock_jwst_data(filename, n_samples=300):
    """Generates mock JWST data embedding the EFT-Lapse Mode E Dipole signal."""
    print(f"Generating mock JWST data with EFT-Lapse V5.1 Dipole (N={n_samples})...")
    np.random.seed(42)
    ra = np.random.uniform(0, 360, n_samples)
    dec = np.degrees(np.arcsin(np.random.uniform(-1, 1, n_samples)))
    z = np.random.uniform(0.5, 1.8, n_samples) # High-z regime
    
    # Calculate distance to Cold Spot
    theta = angular_separation(ra, dec, COLD_SPOT_RA, COLD_SPOT_DEC)
    
    # Inject EFT-Lapse Dipole (Mode E): Max at Shapley (theta=180), Min at Cold Spot (theta=0)
    # A_dip = 2.5 km/s/Mpc (Dipole Amplitude)
    h0_directional = dipole_model(theta, h_iso=70.0, a_dip=-2.5) 
    
    # Generate Distance Modulus with intrinsic scatter (sigma = 0.15 mag)
    mu_true = theoretical_distance_modulus(z, h0_directional, OMEGA_M)
    mu_obs = mu_true + np.random.normal(0, 0.15, n_samples)
    
    df = pd.DataFrame({'RA': ra, 'DEC': dec, 'zHD': z, 'MU_OBS': mu_obs})
    df.to_csv(filename, index=False)
    return df

def analyze_dipole_in_z_bin(df, z_min, z_max, bin_name):
    """Extracts effective H0 and fits the dipole amplitude for a specific redshift bin."""
    subset = df[(df['zHD'] >= z_min) & (df['zHD'] < z_max)].copy()
    if len(subset) < 20:
        return None
    
    # 1. Compute angular distance to Cold Spot
    theta = angular_separation(subset['RA'].values, subset['DEC'].values, 
                               COLD_SPOT_RA, COLD_SPOT_DEC)
    
    # 2. Extract Effective H0 from residuals
    # Delta mu ≈ - (5 / ln(10)) * (Delta H0 / H0)
    mu_background = theoretical_distance_modulus(subset['zHD'].values, h0=70.0, omega_m=OMEGA_M)
    delta_mu = subset['MU_OBS'].values - mu_background
    h0_effective = 70.0 * (1 - (np.log(10) / 5) * delta_mu)
    
    # 3. Fit Dipole Model
    popt, pcov = curve_fit(dipole_model, theta, h0_effective, p0=[70.0, -2.0])
    h_iso, a_dip = popt
    err_h_iso, err_a_dip = np.sqrt(np.diag(pcov))
    
    # 4. Compute Pearson correlation
    corr = np.corrcoef(np.cos(np.radians(theta)), h0_effective)[0, 1]
    
    print(f"--- Epoch: {bin_name} (z={z_min} to {z_max}) | N={len(subset)} ---")
    print(f"  Isotropic Baseline (H_iso) : {h_iso:.2f} ± {err_h_iso:.2f} km/s/Mpc")
    print(f"  Dipole Amplitude (A_dip)   : {abs(a_dip):.2f} ± {err_a_dip:.2f} km/s/Mpc")
    print(f"  Correlation r(H0, cos(th)) : {corr:.3f}")
    
    # Stability Check for EFT-Lapse Falsification
    if abs(a_dip) > 1.0 and corr > 0.3:
        print("  -> RESULT: Mode E Dipole DETECTED. EFT-Lapse Anisotropy confirmed at high-z.")
    else:
        print("  -> RESULT: Dipole NOT significant. Potential falsification of large-scale ∇Ψ.")
    print("")
    return a_dip
